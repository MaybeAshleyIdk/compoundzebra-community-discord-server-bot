bot file tree:
	<data dir>/
	`-- storage.sqlite.db

	<config dir>/
	`-- token.txt
	`-- bot.properties  ((currently nothing planned - maybe in the future))

	<logging dir>/
	`-- read-all*
	`-- latest.log  ->  <YYYY>/latest.log
	`-- <YYYY>/
	    `-- latest.log  ->  <MM>-<DD>.log
	    `-- <MM>-<DD>.log
	    `-- <MM>-<DD>.log.gz

	<runtime dir>/
	`-- console=



usage: cobot [--name=<name>] [@<file>]

    Options:
      --name=<name>  Sets the bot instance name.
                     It is used for the default directory locations and in logs.
                     The default value is "default".

      --data-dir=<path>     Sets the data directory. (default: <default according to the if-else below>)
      --config-dir=<path>   Sets the config directory. (default: <default according to the if-else below>)
      --logging-dir=<path>  Sets the logging directory. (default: <default according to the if-else below>)
      --runtime-dir=<path>  Sets the runtime directory. (default: <default according to the if-else below>)


if effective uid != 0:
	default config dir = $XDG_CONFIG_HOME/cobot/instances/<name>/
	default runtime dir = $XDG_RUNTIME_DIR/cobot/instances/<name>/
	default logging dir = $XDG_STATE_HOME/log/cobot/instances/<name>/
	default data dir = $XDG_DATA_HOME/cobot/instances/<name>/
else:
	default config dir = /etc/cobot/instances/<name>/
	default runtime dir = /run/cobot/instances/<name>/
	default logging dir = /var/log/cobot/instances/<name>/
	default data dir = /var/lib/cobot/instances/<name>/


default <name> is "default"




/etc/systemd/system/cobot.service:
	[Unit]
	Description=CompoundZebra Community Discord Guild Bot
	Documentation=https://github.com/MaybeAshleyIdk/compoundzebra-community-discord-server-bot#readme
	Requires=network-online.target
	After=network-online.target

	[Service]
	Type=exec
	ExecStart=/usr/bin/cobot @%E/cobot-systemd/options.txt
	# ExecStop=/usr/bin/cobot @%E/cobot-systemd/options.txt --console
	Restart=on-failure

	[Install]
	WantedBy=multi-user.target


/etc/systemd/system/cobot@.service:
	[Unit]
	Description=CompoundZebra Community Discord Guild Bot
	Documentation=https://github.com/MaybeAshleyIdk/compoundzebra-community-discord-server-bot#readme
	Requires=network-online.target
	After=network-online.target

	[Service]
	Type=exec
	ExecStart=/usr/bin/cobot @%E/cobot-systemd/options.txt --name=%i @%E/cobot-systemd/instance-options/%i.txt
	Restart=on-failure

	[Install]
	WantedBy=multi-user.target


by default, none of the '%E/cobot-systemd/options.txt' or '%E/cobot-systemd/instance-options/%i.txt' files exist



/etc/systemd/system/cz-community-dc-bot.service:
	[Unit]
	Description=CompoundZebra Community Discord Guild Bot
	Documentation=https://github.com/MaybeAshleyIdk/compoundzebra-community-discord-server-bot#readme
	Requires=network-online.target
	After=network-online.target

	[Service]
	User=<uid>
	Group=<gid>
	Type=exec
	ExecStart=/usr/bin/cobot @%E/cobot-systemd/options.txt @%E/cobot-systemd/instance-options/cz-community-dc-bot.txt --name=cz-community-dc-bot @$HOME/cz-community-dc-bot/options.txt
	Restart=on-failure

	[Install]
	WantedBy=multi-user.target
